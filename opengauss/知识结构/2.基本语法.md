- 范围分区（RANGE）
```
create table table_name(
col1 timestamp
)PARTITION BY RANGE(col1)
(
    PARTITION p0_202401 VALUES LESS THAN('2024-02-01 00:00:00'),
    PARTITION p0_202402 VALUES LESS THAN('2024-03-01 00:00:00'),
    PARTITION p_maxvalue VALUES LESS THAN(MAXVALUE)
);
-- 查询
SELECT * FROM table_name PARTITION(p0_202401);
SELECT * FROM table_name PARTITION FOR ('2024-02-01 00:00:00')
```
- 间隔分区（RANGE-START-END）
  间隔分区是在范围分区的基础上，增加了间隔值的定义
```
create table table_name(
col1 timestamp
)PARTITION BY RANGE(col1)
(
    PARTITION p0_2023 START('2023-01-01 00:00:00') END('2024-01-01 00:00:00') EVERY(INTERVAL '1 MONTH'),
    PARTITION p0_2024 START('2024-01-01 00:00:00') END('2025-01-01 00:00:00') EVERY(INTERVAL '1 MONTH'),
    PARTITION p_maxvalue START('2025-01-01 00:00:00') END(MAXVALUE)
);
```
